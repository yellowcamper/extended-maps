102 * \details This class uses \ref null "extended::null<A>" to define a \c default_value for the \c std::map<A, B> unless it is changed in the constructor, then, it does not save items equal to the \c default_value and all values that are not defined will be defined as the \c default_value.
103 * \note All of the \c map<A, B> operators and functions are left unchanged, so there will not be any conflicts, however, many operations will use new operators to make use of this class.

111 * \note All of the \c map<A, B> operators and functions are left unchanged, so there will not be any conflicts, however, many operations will use new operators to make use of this class.

186 /**
187 * \brief This adds a read only \c operator[] using \c operator>> to "pull" out the value.
188 *
189 * \param [in] input is the location of the desired value.
190 * \return Returns the value of the location if it exists, however, if it does not exist, it will return the \c default_value.
191 */
192 template <class A, class B>
193 B map<A, B>::operator>> (A input) {
194 if ((*this).count(input) > 0) {
195 return (*this).operator[](input);
196 }
197 return default_value;
198 }

200 /**
201 * \brief This adds a pair to the map if and only if the first value is already in use or the second value is not the \c default_value.
202 *
203 * \param [in] input is the pair of the location and the desired value.
204 * \return Returns \c void.
205 */
206 template <class A, class B>
207 void map<A, B>::operator() (std::pair<A, B> input) {
208 if (((*this).count(input.first) > 0) || (input.second != default_value)) {
209 (*this).operator[](input.first) = input.second;
210 }
211 }

213 /**
214 * \brief This adds a pair to the map if and only if the second value is not the \c default_value, if it is \c default_value, then it will also erase the entry.
215 *
216 * \param [in] input is the pair of the location and the desired value.
217 * \return Returns \c void.
218 */
219 template <class A, class B>
220 void map<A, B>::operator<< (std::pair<A, B> input) {
221 if (input.second != default_value) {
222 (*this).operator[](input.first) = input.second;
223 } else if ((*this).count(input.first) > 0) {
224 auto it = (*this).begin();
225 std::advance(it, input.first);
226 (*this).erase(it);
227 }
228 }

230 /**
231 * \brief This removes any \c default_value from the map by erasing the entry.
232 *
233 * \return Returns \c void.
234 */
235 template <class A, class B>
236 void map<A, B>::operator! () {
237 uint64_t temp = 0;
238 for (auto it = (*this).cbegin(); it != (*this).cend(); ) {
239 if ((*it).second == default_value) {
240 temp = std::distance((*this).cbegin(), it);
241 (*this).erase(it);
242 it = (*this).cbegin();
243 std::advance(it, temp);
244 } else {
245 std::advance(it, 1);
246 }
247 }
248 }
